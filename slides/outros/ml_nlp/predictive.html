def spacy_tokenize(text):
    """
    Retorna texto tokenizado em forma de lema, com stop words removidas.
    Também substitui '.' nos tokens por nada para tratar números com e sem ponto.
    Substitui números por 'NUM', urls por 'URL' e e-mails por 'EMAIL'.
    """
    tokens = []
    for t in __NLP__(text):
        if not t.is_stop and not re.match(r"\W+", t.lemma_): # não inclui stop words nem o que não tiver caracter 'word'
            if t.like_num:
                tokens.append("NUM")
            elif t.like_url:
                tokens.append("URL")
            elif t.like_email:
                tokens.append("EMAIL")
            else:
                l = re.sub(r"\d+", "NUM", t.lemma_).replace('.','')
                tokens.append(l)
    
    return tokens

def test_cfl_hyperparams(clf, params_dict, metric='roc_auc', k_fold=10):
    
    grid_clf = GridSearchCV(estimator=clf, 
                            param_grid=params_dict, 
                            scoring=metric, 
                            cv=k_fold, 
                            n_jobs=-1)
    
    grid_clf.fit(X_train, y_train)
    best_clf = grid_clf.best_estimator_
    
    print('Hiperparâmetros do melhor modelo (Dados de treino):', grid_clf.best_params_)
    print(f"\nScore ({metric}) do melhor modelo (Dados de treino): {round(grid_clf.best_score_, 4)}")
    print(f"\nAcurácia preditiva do melhor modelo (Dados de teste): {round(best_clf.score(X_test, y_test)*100, 2)}%")
    
    print("\n\n########################################################\n\n")
    
    y_pred = best_clf.predict(X_test)
    print(classification_report(y_test, y_pred))
    print("\n\n")
    
    report = classification_report(y_test, y_pred, output_dict=True)
    
    return best_clf, grid_clf, report

def grid_results_to_df(grid):
    
    df = pd.concat([pd.DataFrame(grid.cv_results_["params"]), pd.DataFrame(grid.cv_results_)["mean_test_score"]], axis=1)
    
    return df

    'Macro - Precisão': [],
    'Macro - Sensibilidade': [],
    'Macro - F1': [],
    '0 - Precisão': [],
    '0 - Sensibilidade': [],
    '0 - F1': [],
    '1 - Precisão': [],
    '1 - Sensibilidade': [],
    '1 - F1': []
}

    report = reports[algo]
    results['Acurácia'].append(report['accuracy'])
    results['Macro - Precisão'].append(report['macro avg']['precision'])
    results['Macro - Sensibilidade'].append(report['macro avg']['recall'])
    results['Macro - F1'].append(report['macro avg']['f1-score'])
    results['0 - Precisão'].append(report['0']['precision'])
    results['0 - Sensibilidade'].append(report['0']['recall'])
    results['0 - F1'].append(report['0']['f1-score'])
    results['1 - Precisão'].append(report['1']['precision'])
    results['1 - Sensibilidade'].append(report['1']['recall'])
    results['1 - F1'].append(report['1']['f1-score'])

comparison = pd.DataFrame(results).sort_values('Acurácia', ascending=False).reset_index(drop=True)

comparison.to_csv('../DataOutput/NLP_Output/model_comparison.csv', index=False)
